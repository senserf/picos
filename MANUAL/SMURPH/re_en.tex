\newcommand{\putfig}[2]{%
	\begin{figure}[htbp]%
		\begin{center}%
			\input{FIGURES/#1}%
			\usebox{\graph}%
			\vspace{0.75cm}
			\caption{#2\label{fig.#1}}%
		\end{center}%
	\end{figure}%
}%

\newcommand{\puttab}[2]{%
	\begin{table}[htbp]%
		\begin{center}%
			\input{TABLES/#1}%
			\caption{#2\label{tab.#1}}%
		\end{center}%
	\end{table}%
}%

\newcommand{\sectdef}[3]{%
	\ifcase #1%
		\typeout{Section depth #1 undefined; 1 used instead.}%
		\section{#3\label{sect.#2}}%
	\or	\section{#3\label{sect.#2}}%
	\or	\subsection{#3\label{sect.#2}}%
	\or	\subsubsection{#3\label{sect.#2}}%
	\or	\paragraph{#3\label{sect.#2}}%
	\or	\subparagraph{#3\label{sect.#2}}%
	\else%
		\typeout{Section depth #1 undefined; 1 used instead.}%
		\section{#3\label{sect.#2}}%
	\fi%
}%

\def\dllist#1{%
	\begin{list}{[]}%
		{\settowidth{\labelwidth}{#1}%
		 \settowidth{\leftmargin}{#1}%
		 \addtolength{\leftmargin}{2em}%
		 \setlength{\itemsep}{0in}%
		 \setlength{\labelsep}{1em}% 
		}%
\newcommand{\litem}[1]{\item[{\makebox[\labelwidth][l]{##1}}]}%
}

\def\drlist#1{%
	\begin{list}{[]}%
		{\settowidth{\labelwidth}{#1}%
		 \settowidth{\leftmargin}{#1}%
		 \addtolength{\leftmargin}{2em}%
		 \setlength{\itemsep}{0in}%
		 \setlength{\labelsep}{1em}% 
		}%
\newcommand{\ritem}[1]{\item[{\makebox[\labelwidth][r]{##1}}]}%
}
